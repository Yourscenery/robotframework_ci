*** Keywords ***
camera_edit_add
    [Arguments]    ${dicParams}    ${message}=${ADD_CAMERA_SUCCESS}    ${returnKey}=object
    ${url}    set variable    /camera/camera/add
    ${dicAssert}    create dictionary    message=${message}
    ${returnValue}    http_post_return    ${url}    ${dicParams}    ${dicAssert}    ${returnKey}
    ${bIn}=    evaluate    '.' in '${returnKey}'
    ${cameraId}    set variable if    ${bIn}==${false} and '${message}'=='${ADD_CAMERA_SUCCESS}'    ${returnValue}[id]    ${returnValue}
    run keyword if    '${cameraId}'!='${none}' and '${message}'=='${ADD_CAMERA_SUCCESS}'    set suite variable    ${S_ADD_CAMERA_ID}    ${cameraId}
    Comment    #数据库断言--获取每列的值拼接作为where语句
    ${tableFromWhere}    set variable    ${cameraTableName} WHERE id='${cameraId}' and area_id='${dicParams}[areaId]' and device_name='${dicParams}[deviceName]' and device_number='${dicParams}[deviceNumber]'
    run keyword if    '${cameraId}'!='${none}' and '${message}'=='${ADD_CAMERA_SUCCESS}'    assert_response_database_add_or_update    ${camera apikey and dbkey}    ${dicParams}    ${tableFromWhere}
    ...    ELSE    log    ***pass【没新增成功不走数据库断言】***
    [Return]    ${returnValue}

camera_edit_query
    [Arguments]    ${dicParams}    ${total}=0    ${returnKey}=records
    ${url}    set variable    /camera/camera/query
    ${dicAssert}    create dictionary    total=${total}
    ${respKeyVal}    http_post_return    ${url}    ${dicParams}    ${dicAssert}    ${returnKey}
    ${lenRespKeyVal}    get length    ${respKeyVal}
    #获取第一行数据id
    ${cameraId}=    run keyword if    ${lenRespKeyVal}>0    set variable    ${respKeyVal}[0][id]
    ...    ELSE    set variable    ${None}
    #数据库断言------获取每列的值拼接作为where语句
    #${tableWhereStatement}    set variable    id in (${cameraId})
    run keyword if    ${lenRespKeyVal}>0    assert_response_database_query    ${camera apikey and dbkey}    ${respKeyVal}    ${cameraTableName}    id
    ...    ELSE    log    ***pass【没查询数据不走数据库断言】***
    [Return]    ${cameraId}

camera_edit_delete
    [Arguments]    ${dicParams}    ${message}=${DELETE_CAMERA_SUCCESS}
    ${url}    set variable    /camera/camera/delete
    ${dicAssert}    create dictionary    message=${message}
    http_post    ${url}    ${dicParams}    ${dicAssert}
    #数据库断言
    ${ids}    get from dictionary    ${dicParams}    ids[]
    #tableSelectFrom，作为select 的查询字段；
    ${tableSelectFrom}    set variable    select id,deleted from ${cameraTableName}
    run keyword if    '${ids}'!='${none}' and '${message}'=='${DELETE_CAMERA_SUCCESS}'    assert_response_database_delete    ${tableSelectFrom}    id    ${ids}
    ...    ELSE    log    ***pass【没删除成功不走数据库断言】***
