adscreen_edit_add
    [Arguments]    ${dicParams}    ${message}=${ADD_ADSCREEN_SUCCESS}    ${returnKey}=object
    ${url}    set variable    /ad-screen/device/add
    ${dicAssert}    create dictionary    message=${message}
    ${returnValue}    http_post_return    ${url}    ${dicParams}    ${dicAssert}    ${returnKey}
    ${bIn}=    evaluate    '.' in '${returnKey}'
    ${adscreenId}    set variable if    ${bIn}==${false}    ${returnValue}[id]    ${returnValue}
    run keyword if    '${adscreenId}'!='${none}' and '${message}'=='${ADD_ADSCREEN_SUCCESS}'    set suite variable    ${S_ADD_ADSCREEN_ID}    ${adscreenId}
    #数据库断言--获取每列的值拼接作为where语句
    ${tableFromWhere}    set variable    ${adscreenTableName} WHERE id='${adscreenId}' and device_name='${dicParams}[deviceName]' and card_number='${dicParams}[cardNumber]'
    run keyword if    '${adscreenId}'!='${none}' and '${message}'=='${ADD_ADSCREEN_SUCCESS}'    assert_response_database_add_or_update    ${adscreen apikey and dbkey}    ${dicParams}    ${tableFromWhere}
    ...    ELSE    log    ***pass【没新增成功不走数据库断言】***
    [Return]    ${adscreenId}

adscreen_edit_delete
    [Arguments]    ${dicParams}    ${message}=${DELETE_ADSCREEN_SUCCESS}
    ${url}    set variable    /adscreen/adscreen/delete
    ${dicAssert}    create dictionary    message=${message}
    http_post    ${url}    ${dicParams}    ${dicAssert}
    #数据库断言
    ${ids}    get from dictionary    ${dicParams}    ids[]
    #tableSelectFrom，作为select 的查询字段；
    ${tableSelectFrom}    set variable    select id,deleted from ${adscreenTableName}
    run keyword if    '${ids}'!='${none}' and '${message}'=='${DELETE_ADSCREEN_SUCCESS}'    assert_response_database_delete    ${tableSelectFrom}    id    ${ids}
    ...    ELSE    log    ***pass【没删除成功不走数据库断言】***
