spon_edit_add
    [Arguments]    ${dicParams}    ${success}=${ADD_SPON_SUCCESS}    ${returnKey}=object
    ${url}    set variable    /rf-spon/terminals
    ${dicAssert}    create dictionary    success=${success}
    ${sponId}    http_post_return    ${url}    ${dicParams}    ${dicAssert}    ${returnKey}
    run keyword if    '${sponId}'!='${none}' and '${success}'=='${ADD_SPON_SUCCESS}'    set suite variable    ${S_ADD_SPON_ID}    ${sponId}
    #数据库断言--获取每列的值拼接作为where语句
    ${tableFromWhere}    set variable    ${sponTableName} WHERE id='${sponId}' and tname='${dicParams}[tname]' and tid='${dicParams}[tid]'
    run keyword if    '${sponId}'!='${none}' and '${success}'=='${ADD_SPON_SUCCESS}'    assert_response_database_add_or_update    ${spon apikey and dbkey}    ${dicParams}    ${tableFromWhere}
    ...    ELSE    log    ***pass【没新增成功不走数据库断言】***
    [Return]    ${sponId}

spon_edit_delete
    [Arguments]    ${dicParams}    ${message}=${DELETE_SPON_SUCCESS}
    ${url}    set variable    /spon/spon/delete
    ${dicAssert}    create dictionary    message=${message}
    http_post    ${url}    ${dicParams}    ${dicAssert}
    #数据库断言
    ${ids}    get from dictionary    ${dicParams}    ids[]
    #tableSelectFrom，作为select 的查询字段；
    ${tableSelectFrom}    set variable    select id,deleted from ${sponTableName}
    run keyword if    '${ids}'!='${none}' and '${message}'=='${DELETE_SPON_SUCCESS}'    assert_response_database_delete    ${tableSelectFrom}    id    ${ids}
    ...    ELSE    log    ***pass【没删除成功不走数据库断言】***
